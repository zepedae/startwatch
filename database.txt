CREATE DATABASE startwatch;

#users:
CREATE TABLE users (id INT AUTO_INCREMENT PRIMARY KEY NOT NULL, username VARCHAR(25) NOT NULL, password VARCHAR(255) NOT NULL);

#watches:
CREATE TABLE watches (id INT AUTO_INCREMENT PRIMARY KEY NOT NULL, user_id INT NOT NULL, username VARCHAR(25) NOT NULL, watch_name VARCHAR(25) NOT NULL, date DATETIME NOT NULL DEFAULT NOW(), time_elapsed TIME, FOREIGN KEY (user_id) REFERENCES users(id))
